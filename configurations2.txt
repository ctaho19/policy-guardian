WITH LATEST_TIMESTAMP AS (
    SELECT MAX(SF_LOAD_TIMESTAMP) AS LATEST_TS
    FROM OZONE_QUARTERLY_EXAMINER_REVIEW_TOTAL_COUNTS_REPORT_V01_V4
),
BASE_RANKED_RECORDS AS (
    SELECT *,
    ROW_NUMBER() OVER (
        PARTITION BY ACCOUNT_NUMBER, REGION, ASV, RESOURCE_TYPE 
        ORDER BY BATCH_ID DESC
    ) AS RANK
    FROM OZONE_QUARTERLY_EXAMINER_REVIEW_TOTAL_COUNTS_REPORT_V01_V4
    WHERE BATCH_ID = (SELECT MAX(BATCH_ID) FROM OZONE_QUARTERLY_EXAMINER_REVIEW_TOTAL_COUNTS_REPORT_V01_V4)
    AND CLOUD_PROVIDER = 'AWS'
    AND ROLE_TYPE IN ('', 'N/A', 'MACHINE')
    AND CONTROL_ID = '<your_control_id>'
    AND LOWER(REGION) IN ('ca-central-1', 'eu-west-1', 'eu-west-2', 'us-east-1', 'us-east-2', 'us-west-2', 'global')
),
INITIAL_COUNTS AS (
    SELECT 
        'Initial Data Check' as CHECK_POINT,
        COUNT(*) as TOTAL_RECORDS,
        COUNT(DISTINCT CONCAT(ASV, ACCOUNT_NUMBER, REGION, RESOURCE_TYPE)) as UNIQUE_GROUPS,
        SUM(INVENTORY_COUNT) as TOTAL_INVENTORY,
        SUM(TOTAL_EVALUATIONS_COUNT) as TOTAL_EVALUATIONS
    FROM BASE_RANKED_RECORDS
    WHERE RANK = 1
),
COMPONENT_EVALUATIONS AS (
    SELECT 
        ASV,
        ACCOUNT_NUMBER,
        REGION,
        RESOURCE_TYPE,
        CONTROL_COMPONENT,
        COUNT(*) as RECORDS_PER_COMPONENT,
        SUM(TOTAL_EVALUATIONS_COUNT) as COMPONENT_EVAL_COUNT,
        SUM(INVENTORY_COUNT) as COMPONENT_INVENTORY
    FROM BASE_RANKED_RECORDS
    WHERE RANK = 1
    GROUP BY 
        ASV,
        ACCOUNT_NUMBER,
        REGION,
        RESOURCE_TYPE,
        CONTROL_COMPONENT
),
COMPONENT_STATS AS (
    SELECT 
        'Component Statistics' as CHECK_POINT,
        COUNT(DISTINCT CONTROL_COMPONENT) as UNIQUE_COMPONENTS,
        COUNT(DISTINCT CONCAT(ASV, ACCOUNT_NUMBER, REGION, RESOURCE_TYPE)) as UNIQUE_GROUPS,
        SUM(COMPONENT_EVAL_COUNT) as TOTAL_COMPONENT_EVALS,
        SUM(COMPONENT_INVENTORY) as TOTAL_COMPONENT_INVENTORY,
        AVG(COMPONENT_EVAL_COUNT) as AVG_EVALS_PER_COMPONENT
    FROM COMPONENT_EVALUATIONS
)
-- Show initial counts
SELECT * FROM INITIAL_COUNTS
UNION ALL
SELECT * FROM COMPONENT_STATS
UNION ALL
-- Show sample of actual data
SELECT 
    'Sample Data' as CHECK_POINT,
    ce.ASV as TOTAL_RECORDS,
    ce.ACCOUNT_NUMBER as UNIQUE_GROUPS,
    ce.CONTROL_COMPONENT as TOTAL_INVENTORY,
    ce.COMPONENT_EVAL_COUNT as TOTAL_EVALUATIONS
FROM COMPONENT_EVALUATIONS ce
LIMIT 5;
