   SELECT RESOURCE_NAME, ROLE_TYPE, CREATE_DATE, LENGTH(RESOURCE_NAME) AS name_length
   FROM EIAM_DB.PHDP_CYBR_IAM.IDENTITY_REPORTS_CONTROLS_VIOLATIONS_STREAM_V2
   WHERE RESOURCE_NAME LIKE '%fs-datainsghts-ui%'
   ORDER BY CREATE_DATE DESC;

   SELECT 
       r.RESOURCE_ID,
       r.AMAZON_RESOURCE_NAME AS role_arn,
       v.RESOURCE_NAME AS violation_arn,
       v.ROLE_TYPE,
       LENGTH(r.AMAZON_RESOURCE_NAME) AS role_length,
       LENGTH(v.RESOURCE_NAME) AS violation_length
   FROM EIAM_DB.PHDP_CYBR_IAM.IDENTITY_REPORTS_IAM_RESOURCE r
   LEFT JOIN EIAM_DB.PHDP_CYBR_IAM.IDENTITY_REPORTS_CONTROLS_VIOLATIONS_STREAM_V2 v
       ON r.AMAZON_RESOURCE_NAME = v.RESOURCE_NAME
   WHERE r.AMAZON_RESOURCE_NAME LIKE '%fs-datainsghts-ui%'
   ORDER BY v.CREATE_DATE DESC;
