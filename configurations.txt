SELECT 
    CONTROL_ID, 
    SUM(TOTAL_EVALUATIONS_COUNT) AS TOTAL_EVALUATED_COUNT, 
    SUM(INVENTORY_COUNT) as INVENTORY_COUNT
FROM CLCN_DB.PHDP_CLOUD.OZONE_QUARTERLY_EXAMINER_REVIEW_TOTAL_COUNTS_REPORT_V01_V4
WHERE RESOURCE_TYPE = 'AWS::EC2::Instance'
AND BATCH_ID = (SELECT MAX(BATCH_ID) FROM CLCN_DB.PHDP_CLOUD.OZONE_QUARTERLY_EXAMINER_REVIEW_TOTAL_COUNTS_REPORT_V01_V4)
AND SF_LOAD_TIMESTAMP >= '2024-12-18'
AND CONTROL_ID = 'your_control_id'  -- Replace with your specific control ID
GROUP BY CONTROL_ID
ORDER BY CONTROL_ID;
